# vim

# Search in a file

# enter search mode

/<pattern>       # then Enter

# navigate through results

n or N

# Navigate in a file

# go to beginning of next word (+ shift to include symbols)

w

# go to end of next word (+ shift to include symbols)

e

# go to beginning of previous word (+ shift to include symbols)

b

# go to the end of the line

$

# go to the beginning of the line

:0

# go to nth line

:n

# execute a command

:!command %
# in this context, "%" is the current filename

# open a file with vim at nth line

vim +n file

# open two files in split mode

vim -O file1 file2
# then ctrl+g + left / right arrow to switch between files

# go to the last line

:$ or G

# go to next paragraph

}

# go to previous paragraph

{

# Edit a file

# Enter insert mode

i

# Enter insert mode at a new line

o

# Enter insert mode just after current cursor position

a

# delete current character

x

# delete word from current position

dw

# delete current line

dd

# delete end of line from current position

D

# copy current line

yy

# copy 4 lines from current one

y4y

# cut current line

dd

# paste under current cursor position

p

# search and replace

:%s/toto/tata/g

# search and replace in given line range

:10,20s/toto/tata/g

# comment out lines 10 to 15

:10;15s/^/#

# join current line with next

j

# Misc

# execute a given command n times

<n><command>

# redo last command

.

# undo last action

u

# execute a bash command

:!command
