# misc unix stuff

# Command chaining

# execute two commands sequentially

command1; command2

# execute command2 only if command1 succeeded

command1 && command2

# execute command2 only if command1 failed

command1 || command2

# Redirections

# redirect stdout to a file

echo test > output

# redirect stdout to a file - append mode

echo test >> output

# redirect stderr to a file

ls error 2> output

# redirect stdin and stderr to different files

command > out.txt 2> error.txt

# redirect both stdin and stderr to the same file

command &> output.txt

# nevermind the output of that command

command > /dev/null

# redirect a file to stdin

say < file_to_read.txt

# space when no space allowed

${IFS}
${SPACE}

# launch a command in the background

command &

# get the output status of last command

$?

# get the last argument of the last command

$_

# display the size of files or folders in human-readable format

ls -lh

# cat with line numbers

cat -n filename

# follow the output of several log files at the same time

tail -f file1 file2
