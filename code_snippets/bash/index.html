# do something when a webpage changes

#!/bin/bash

FOLDER="" # path to the folder for outputing
URL="https://test.com"    # URL to check
TIMEOUT="30"

while true; do
		curl $URL >| $FOLDER/new.txt

		cp $FOLDER/new.txt $FOLDER/old.txt

		res=$(diff $FOLDER/new.txt $FOLDER/old.txt)

		if [ "0" != "${#res}" ]
		then
			echo $res;
			osascript -e "display notification \"changement détecté\" with title \"Alerte\"" # this part only works for OS X
		fi

		cp $FOLDER/new.txt $FOLDER/old.txt

		sleep $TIMEOUT;
done
